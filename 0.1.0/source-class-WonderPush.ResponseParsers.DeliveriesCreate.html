  /**

   * @return $this

    return $this;
  }

  /**
   * @return string
   */
  public function getCampaignId() {
    return $this->campaignId;
  }

  /**
   * @param string $campaignId
   * @return $this

  /**
   * @return string

    $this->notificationId = $notificationId;
    return $this;

}
